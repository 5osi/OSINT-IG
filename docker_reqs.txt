FROM python:3.9.2-alpine3.13 as build
WORKDIR /wheels
RUN apk add --no-cache \
    ncurses-dev \
    build-base
COPY docker_reqs.txt /opt/osint-ig/requirements.txt
RUN pip3 wheel -r /opt/osint-ig/requirements.txt


FROM python:3.9.2-alpine3.13
WORKDIR /home/osint-ig
RUN adduser -D osint-ig

COPY --from=build /wheels /wheels
COPY --chown=osint-ig:osint-ig requirements.txt /home/osintgram/
RUN pip3 install -r requirements.txt -f /wheels \
  && rm -rf /wheels \
  && rm -rf /root/.cache/pip/* \
  && rm requirements.txt

COPY --chown=osint-ig:osint-ig src/ /home/osint-ig/src
COPY --chown=osint-ig:osint-ig main.py /home/osint-ig/
COPY --chown=osint-ig:osint-ig config/ /home/osint-ig/config
USER osint-ig

ENTRYPOINT ["python", "main.py"]
